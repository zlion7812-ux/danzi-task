<script>
    let tasksData = [];
    let defeatedIds = [];
    let portalAvailable = false;

    async function loadGameData() {
        // 获取积分/连胜
        let statusRes = await fetch('/status');
        let status = await statusRes.json();
        document.getElementById('pointsDisplay').innerText = status.points;
        document.getElementById('streakDisplay').innerText = status.streak;

        // 获取任务列表 + 完成状态（通过现有接口组合）
        let tasksRes = await fetch('/tasks_list');
        let tasksJson = await tasksRes.json();
        tasksData = tasksJson.tasks;
        defeatedIds = tasksJson.completed_ids || [];
        renderMap();
    }

    function renderMap() {
        const container = document.getElementById('taskNodes');
        container.innerHTML = '';
        const total = tasksData.length;
        document.getElementById('totalTasks').innerText = total;
        const completedCount = defeatedIds.length;
        document.getElementById('progressCount').innerText = completedCount;
        const percent = (completedCount / total) * 100;
        document.getElementById('progressFill').style.width = percent + '%';

        tasksData.forEach((task, idx) => {
            const isCompleted = defeatedIds.includes(task.id);
            const node = document.createElement('div');
            node.className = `node-card ${isCompleted ? 'completed' : ''}`;
            node.onclick = () => fightMonster(task.id);
            node.innerHTML = `
                <div class="node-icon">${task.icon}</div>
                <div class="node-name">${task.name}</div>
                <div class="node-status">${isCompleted ? '✅ 已完成' : '⚔️ 可讨伐'}</div>
            `;
            container.appendChild(node);
            if (idx !== tasksData.length - 1) {
                const lineSpan = document.createElement('span');
                lineSpan.className = 'connector-line';
                lineSpan.innerHTML = ' ➡️ ';
                container.appendChild(lineSpan);
            }
        });

        // 传送门条件：完成 50% 以上任务
        if (!portalAvailable && completedCount >= Math.ceil(total * 0.5)) {
            portalAvailable = true;
            showPortal();
        }
    }

    async function fightMonster(taskId) {
        let resp = await fetch('/fight', {
            method: 'POST',
            headers: {'Content-Type': 'application/json'},
            body: JSON.stringify({task_id: taskId})
        });
        let data = await resp.json();
        if (data.error) {
            document.getElementById('lootDisplay').innerHTML = '⚠️ ' + data.error;
            return;
        }
        document.getElementById('lootDisplay').innerHTML = `⚔️ 击败 ${data.monster_name}！<br>🎁 获得 ${data.base_points} 积分 ✨ ${data.loot_effect}`;
        loadGameData(); // 刷新地图状态
    }

    function showPortal() {
        const portalDiv = document.getElementById('portalArea');
        portalDiv.innerHTML = `<div class="node-card portal-card" style="display:inline-block; padding:12px 30px;" onclick="openPortal()">
                                    🌀 神秘传送门 🌀<br>✨ 点击进入异界 ✨
                                </div>`;
    }

    async function openPortal() {
        let resp = await fetch('/portal', {method: 'POST'});
        let data = await resp.json();
        if (data.status === 'ok') {
            alert('✨ 你进入了秘境，获得了彩蛋任务！✨');
            location.reload();
        } else {
            alert(data.error);
        }
    }

    function switchTab(tab) {
        if (tab === 'shop') window.location.href = '/shop';
        if (tab === 'points') window.location.href = '/points';
        if (tab === 'quest') window.location.reload();
    }

    loadGameData();
</script>
